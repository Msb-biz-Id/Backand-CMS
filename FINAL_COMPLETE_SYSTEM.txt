🔄 DATA FLOW - MVC ARCHITECTURE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

User Request → Router → Controller → Model → Database
                                              ↓
                                         ← Data (Array)
                                              ↓
                                         View (Render)
                                              ↓
                                         HTML Output

Example Post Display:
━━━━━━━━━━━━━━━━━━━━
Request: /post/my-awesome-article
   ↓
PostController.php → show($slug)
   ↓
Post.php → getBySlug($slug)
   ↓
Database → SELECT * FROM cms_posts WHERE slug = ?
   ↓
← $post = ['id' => 1, 'title' => '...', 'content' => '...']
   ↓
View: frontend/posts/show.php
   ↓
<?php echo htmlspecialchars($post['title']); ?>  // DINAMIS!
<?php echo $post['content']; ?>                  // DINAMIS!
